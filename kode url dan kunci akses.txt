// MESIN KOMPRESI GAMBAR (Tetap dipertahankan agar database Supabase tidak cepat penuh)
function kompresGambar(file) {
    return new Promise((resolve) => {
        const reader = new FileReader();
        reader.readAsDataURL(file);
        reader.onload = (event) => {
            const img = new Image();
            img.src = event.target.result;
            img.onload = () => {
                const canvas = document.createElement('canvas');
                const MAX_WIDTH = 800; 
                let width = img.width;
                let height = img.height;
                if (width > MAX_WIDTH) {
                    height *= MAX_WIDTH / width;
                    width = MAX_WIDTH;
                }
                canvas.width = width;
                canvas.height = height;
                const ctx = canvas.getContext('2d');
                ctx.drawImage(img, 0, 0, width, height);
                // Kita kompres ke 0.6 (60%) agar lebih ringan di Supabase
                resolve(canvas.toDataURL('image/jpeg', 0.6)); 
            };
        };
    });
}

async function uploadMultipleFiles() {
    const fileInput = document.getElementById('file_evidence');
    const files = fileInput.files;
    const statusEl = document.getElementById('statusUpload');
    
    if (files.length === 0) return null;

    if (statusEl) statusEl.innerText = `⏳ Mengupload ${files.length} foto...`;
    let daftarUrl = [];

    for (let i = 0; i < files.length; i++) {
        const file = files[i];
        const fileExt = file.name.split('.').pop();
        const fileName = `${Date.now()}_${i}.${fileExt}`;
        const filePath = `evidence/${fileName}`;

        // Proses Upload per file
        const { data, error } = await _supabase.storage
            .from('mhc_storage') 
            .upload(filePath, file);

        if (error) {
            console.error("Gagal upload file ke-" + i, error);
            continue;
        }

        const { data: urlData } = _supabase.storage
            .from('mhc_storage')
            .getPublicUrl(filePath);
        
        daftarUrl.push(urlData.publicUrl);
    }

    if (statusEl) statusEl.innerText = "✅ Foto berhasil terupload!";
    return daftarUrl.join(","); // Hasilnya: "url1,url2,url3"
}

async function previewFotos() {

async function simpanDraft(event) {
    const elNotif = document.getElementById('nomor_notif');
    if (!elNotif || !elNotif.value) return alert("Pilih nomor notif dulu!");

    const notif = elNotif.value;
    const tgl = new Date().toLocaleDateString('id-ID');
    const jam = new Date().toLocaleTimeString('id-ID', { hour: '2-digit', minute: '2-digit' });

    // 1. Ambil data asli dari memori (cloudData) untuk digabungkan nanti
    const dataLama = cloudData.find(r => String(r.nomor_notif) === String(notif));
    if (!dataLama) return alert("Data tidak ditemukan di memori!");

    // 2. Fungsi Pembantu: Menggabungkan Riwayat Lama di DB dengan Ketikan Baru di Form
    const gabungTeks = (idElemen, teksLamaDatabase) => {
        const inputBaru = document.getElementById(idElemen)?.value.trim() || "";
        const riwayatLama = (teksLamaDatabase || "").toString().trim();

        // Jika mekanik mengetik sesuatu yang baru
        if (inputBaru !== "" && inputBaru !== "-") {
            const separator = (riwayatLama === "" || riwayatLama === "-") ? "" : riwayatLama + "\n\n";
            return `${separator}--- UPDATE ${tgl} ${jam} ---\n${inputBaru}`;
        }
        // Jika kotak input kosong, kirimkan data lama apa adanya (agar tidak hilang)
        return riwayatLama || "-";
    };

    // 3. Proses Penggabungan untuk semua kolom laporan
    const analisaFinal = gabungTeks('analisa', dataLama.analisa);
    const tindakanFinal = gabungTeks('tindakan_pekerjaan', dataLama.tindakan_pekerjaan);
    const resumeFinal = gabungTeks('resume', dataLama.resume);
    const kendalaFinal = gabungTeks('kendala', dataLama.kendala);
    const prog = document.getElementById('progRange')?.value || 0;

    // 4. LOGIKA STATUS (PENTING: Cek langsung ke elemen input di layar)
    const inputKendalaSekarang = document.getElementById('kendala')?.value.trim() || "";
    let statusFinal = "Inprogres"; 

    // Jika di layar kotak kendala ada isinya, status jadi Pending
    if (inputKendalaSekarang !== "" && inputKendalaSekarang !== "-") {
        statusFinal = "Pending";
    } else {
        // Jika kotak kendala kosong, status balik ke Inprogres
        statusFinal = "Inprogres";
    }

    // 5. Eksekusi Update ke Supabase
    const btn = event.currentTarget;
    btn.disabled = true;
    btn.innerText = "⏳ Menyimpan...";

    try {
        // --- GANTI BAGIAN UPLOAD LAMA DENGAN INI ---
        const linkFotoBaru = await uploadMultipleFiles();
        
        // Gabungkan dengan link lama yang sudah ada di database
        let evidenceFinal = dataLama.foto_pekerjaan || "";
        if (linkFotoBaru) {
            evidenceFinal = (evidenceFinal === "" || evidenceFinal === "-") 
                ? linkFotoBaru 
                : evidenceFinal + "," + linkFotoBaru;
        }
        const { error } = await _supabase
            .from('pekerjaan')
            .update({
                analisa: analisaFinal,
                tindakan_pekerjaan: tindakanFinal,
                resume: resumeFinal,
                kendala: kendalaFinal,
                progres: parseInt(prog) || 0,
                foto_pekerjaan: evidenceFinal, // Update link foto yang sudah digabung
                status: statusFinal
            })
            .eq('nomor_notif', notif);

        if (error) throw error;

        // 6. REFRESH DATA & TAMPILAN
        // Ambil data terbaru dari internet agar memori (cloudData) terupdate
        await fetchDataDariSupabase(); 
        
        // Update list/daftar di bagian atas
        if (typeof filterOutstanding === "function") filterOutstanding();

        alert(`✅ BERHASIL SIMPAN!\nStatus: ${statusFinal}`);

        // Panggil kembali pilihNotif agar form bersih dan Banner Status berubah warna
        pilihNotif(notif); 

    } catch (err) {
        console.error("Gagal simpan:", err);
        alert("Gagal simpan: " + err.message);
    } finally {
        btn.disabled = false;
        btn.innerText = "📝 SIMPAN DRAFT";
    }
}

// --- 2. FILTER BOX ANTREAN (Daftar Kerja yang Belum Kelar) ---
function filterOutstanding() {
    const plant = document.getElementById('monPlant')?.value;
    const area = document.getElementById('monArea')?.value;
    const status = document.getElementById('monStatus')?.value;
    const listDiv = document.getElementById('listOutstanding');

    if (!listDiv) return;

    if (!cloudData || cloudData.length === 0) {
        listDiv.innerHTML = '<p style="text-align:center; padding:20px; color:#94a3b8; font-size:12px;">🔄 Menghubungkan ke Supabase...</p>';
        return;
    }

    let filtered = cloudData.filter(row => {
        const rowPlant = String(row.plant || "");
        const rowArea = String(row.area || "");
        const rowStatus = String(row.status || "").toLowerCase().trim();
        const filterStatus = String(status || "").toLowerCase().trim();

        const matchPlant = !plant || rowPlant === plant;
        const matchArea = !area || rowArea === area;
        
        let matchStatus = false;
        if (!filterStatus) {
            matchStatus = (rowStatus !== "finish" && rowStatus !== "closed");
        } else {
            matchStatus = (rowStatus === filterStatus);
        }

        return matchPlant && matchArea && matchStatus;
    });

    if (filtered.length === 0) {
        listDiv.innerHTML = '<p style="text-align:center; padding:20px; color:#94a3b8; font-size:12px;">Tidak ada antrean pekerjaan.</p>';
        return;
    }

    listDiv.innerHTML = filtered.map(row => {
        const nNotif = row.nomor_notif || "Tanpa Notif";
        const progres = row.progres || 0;
        const statusDB = row.status || "Outstanding";
        const kendala = row.kendala || "";

        // --- INI PERUBAHANNYA: LOGIKA STABIL MAS ABDUL ---
        let sLabel = statusDB;
        let color = '#004a99'; // Default Pusri Blue

        if (statusDB.toLowerCase() === 'finish' || statusDB.toLowerCase() === 'closed') {
            sLabel = "Closed"; 
            color = '#16a34a'; // Hijau Sukses
        } else if (statusDB.toLowerCase() === "waiting approval" || progres >= 100) {
            sLabel = "Verifikasi"; 
            color = '#16a34a'; // Hijau Sukses
        } else if (kendala && kendala !== "-" && kendala.trim() !== "") {
            sLabel = "Pending"; 
            color = '#ef4444'; // Merah Kendala
        } else if (progres > 0) {
            sLabel = "In Progres"; 
            color = '#004a99'; // Biru Inprogres
        } else {
            sLabel = "Outstanding"; 
            color = '#ea580c'; // Oranye Awal
        }

        return `
            <div class="out-item" style="border-bottom:1px solid #eee; padding:12px; display:flex; justify-content:space-between; align-items:center;">
                <div onclick="pilihNotif('${nNotif}')" style="flex:1; cursor:pointer;">
                    <span style="font-weight:bold; color:#004a99">${nNotif}</span><br>
                    <small>${row.plant || '-'} | ${row.area || '-'} | ${row.nomor_item || 'No Item'}</small>
                </div>
                <div style="text-align:right;">
                    <div style="font-weight:bold; color:#333; font-size:12px; margin-bottom:4px;">${progres}%</div>
                    <button class="btn-gold" 
                            style="padding:4px 8px; font-size:10px; border-radius:4px; border:1px solid #000; cursor:pointer;" 
                            onclick="lihatStruk('${row.evidence_form}')">
                        👁️ STRUK
                    </button>
                    <span class="mp-chip" style="background:${color}; color:white; border:none; font-size:10px; padding:2px 8px; border-radius:10px; margin-left:5px; display:inline-block; min-width:60px; text-align:center;">
                        ${sLabel}
                    </span>
                </div>
            </div>
        `;
    }).join('');
}

function renderHist() {
    const histBody = document.getElementById('histBody');
    if (!histBody) return;

    // Ambil nilai filter dari inputan user
    const fPlant = document.getElementById('fPlant').value;
    const fArea = document.getElementById('fArea').value;
    const fItem = document.getElementById('fItem').value.toLowerCase();

    // Filter data yang akan ditampilkan di tabel arsip
    let filtered = cloudData.filter(row => {
        const matchPlant = !fPlant || (row.plant && String(row.plant) === fPlant);
        const matchArea = !fArea || (row.area && String(row.area) === fArea);
        const matchItem = !fItem || (row.nomor_item && String(row.nomor_item).toLowerCase().includes(fItem));
        return matchPlant && matchArea && matchItem;
    });

    if (filtered.length === 0) {
        histBody.innerHTML = '<tr><td colspan="7" style="text-align:center; padding:20px;">Data tidak ditemukan</td></tr>';
        return;
    }

    // Masukkan data ke dalam tabel
    histBody.innerHTML = filtered.map(row => `
        <tr>
            <td>${row.timestamp || row.waktu || '-'}</td>
            <td><b>${row.nomor_notif || '-'}</b></td>
            <td>
                <small>${row.plant || '-'} / ${row.area || '-'}</small><br>
                <b>${row.nomor_item || '-'}</b>
            </td>
            <td>
                <div class="mp-chip">${row.penerima || '-'}</div><br>
                <small>Team: ${row.manpower || '-'}</small>
            </td>
            <td>
                <small>${row.tindakan || '-'}</small>
                <div class="mini-prog-bg">
                    <div class="mini-prog-fill" style="width:${row.progres}%; background:${getStatusColor(row.status)}"></div>
                </div>
            </td>
            <td><span class="mp-chip" style="background:${getStatusColor(row.status)}; color:white; border:none">${row.status || 'Outstanding'}</span></td>
            <td class="no-print">
                <button class="btn-blue" style="padding:5px; font-size:10px; margin-bottom:5px; width:100%; display:block;" onclick="pilihNotif('${row.nomor_notif}')">EDIT / DETAIL</button>
        </tr>
    `).join('');
}

function pilihNotif(notif) {
    const data = cloudData.find(r => String(r.nomor_notif) === String(notif));
    if (data) {
        // 1. DATA HEADER (STARK/TETAP)
        document.getElementById('nomor_notif').value = data.nomor_notif || "";
        document.getElementById('nomor_item').value = data.nomor_item || "";
        document.getElementById('plant').value = data.plant || "";
        document.getElementById('area').value = data.area || "";
        document.getElementById('deskOrder').value = data.deskripsi_order || data.deskripsi || "";

        
        // 3. MAPPING HISTORY & PROTEKSI (Hanya yang perlu Riwayat & Kunci)
        const mappingHistory = [
            { id: 'analisa', val: data.analisa },
            { id: 'tindakan_pekerjaan', val: data.tindakan_pekerjaan },
            { id: 'manpower_history', val: data.manpower },
            { id: 'penerima_history', val: data.penerima_order },
            { id: 'kendala', val: data.kendala },
            { id: 'resume', val: data.resume },
            { id: 'evidence', val: data.foto_pekerjaan || "" }
        ];
        mappingHistory.forEach(item => {
        const el = document.getElementById(item.id);
        if (el) {
        const teksLama = (item.val || "").toString().trim();
        
        // 1. JIKA DATA KOSONG (Baru atau belum ada riwayat)
        if (teksLama === "" || teksLama === "-" || !teksLama) {
            el.value = ""; // Biarkan kosong agar bersih
            el.placeholder = "Belum ada riwayat, silakan tulis di sini...";
            el.dataset.oldLength = 0;
        } 
        // 2. JIKA SUDAH ADA DATA (Tampilkan data lama tanpa paksa tambah label UPDATE)
        else {
            // Tampilkan data murni dari database
            el.value = teksLama; 
            
            // Simpan panjang teks lama untuk proteksi
            el.dataset.oldLength = teksLama.length;
        }
    }
});

        // 4. UPDATE PROGRESS & UI (Lanjutan yang hilang)
        const progVal = parseInt(data.progres) || 0;
        document.getElementById('progRange').value = progVal;
        updateProgUI();
        // 4. UPDATE PENERIMA (INPUT AKTIF)
        if (document.getElementById('penerima')) {
            document.getElementById('penerima').value = data.penerima_order || "";
        }


        const banner = document.getElementById('statusBanner');
        if (banner) {
            banner.innerText = "STATUS: " + (data.status || "OUTSTANDING").toUpperCase();
            banner.style.background = getStatusColor(data.status);
        }

        const boxAnalisa = document.getElementById('analisa');
        if (boxAnalisa) {
            boxAnalisa.focus();
            boxAnalisa.setSelectionRange(boxAnalisa.value.length, boxAnalisa.value.length);
        }
    } // Penutup if (data)
} // Penutup function pilihNotif

// --- FUNGSI UPDATE UI PROGRES & BANNER (YANG TERHAPUS) ---
function updateProgUI() {
    const val = document.getElementById('progRange').value;
    const progText = document.getElementById('progText');
    const mainBar = document.getElementById('mainBar');
    const banner = document.getElementById('statusBanner');

    // 1. Update teks persentase
    if(progText) progText.innerText = val + "%";
    
    // 2. Update Progress Bar (Warna & Lebar)
    if(mainBar) {
        mainBar.style.width = val + "%";
        mainBar.innerText = val + "%";
        
        // Perubahan Warna Bar
        if (val < 50) {
            mainBar.style.background = "#dc2626"; // Merah
        } else if (val < 100) {
            mainBar.style.background = "#f59e0b"; // Kuning/Oranye
        } else {
            mainBar.style.background = "#16a34a"; // Hijau
        }
    }

    // 3. Update Status Banner secara Otomatis
    if(banner) {
        if (val == 0) {
            banner.innerText = "STATUS: OUTSTANDING";
            banner.style.background = "#dc2626"; // Merah
        } else if (val == 100) {
            banner.innerText = "STATUS: FINISH";
            banner.style.background = "#16a34a"; // Hijau
        } else {
            banner.innerText = "STATUS: INPROGRES";
            banner.style.background = "#2563eb"; // Biru
        }
    }
}
function getStatusColor(status) {
    const s = (status || "").toLowerCase();
    if (s === 'outstanding') return "#dc2626"; // Merah
    if (s === 'inprogres') return "#004a99";   // Biru
    if (s === 'pending') return "#ff8c00";     // Oranye (Safety Pusri)
    if (s === 'finish') return "#16a34a";      // Hijau
    return "#64748b"; 
}

function lihatStruk(data) {
    if (!data || data === "undefined" || data === "" || data === "-") {
        alert("⚠️ Data link struk kosong.");
        return;
    }

    const modal = document.getElementById('modalStruk');
    const imgDisplay = document.getElementById('imgStruk');
    let srcTujuan = data;

    // Konversi Link Drive ke Direct Link
    if (data.includes("drive.google.com")) {
        let fileId = "";
        if (data.includes("/d/")) {
            fileId = data.split('/d/')[1].split('/')[0];
        } else if (data.includes("id=")) {
            fileId = data.split('id=')[1].split('&')[0];
        }
        if (fileId) {
            srcTujuan = "https://docs.google.com/uc?export=view&id=" + fileId;
        }
    }

    modal.style.display = "flex";
    imgDisplay.src = "https://cdnjs.cloudflare.com/ajax/libs/galleriffic/2.0.1/css/loader.gif"; 

    const tempImg = new Image();
    tempImg.src = srcTujuan;

    tempImg.onload = function() {
        imgDisplay.src = srcTujuan;
    };

    tempImg.onerror = function() {
        modal.style.display = "none";
        // JIKA GAGAL DIMUAT (Biasanya karena proteksi Drive), LANGSUNG BUKA TAB BARU
        if (confirm("⚠️ Gambar tidak bisa tampil langsung di sini karena proteksi Google Drive.\n\nBuka gambar di tab baru?")) {
            window.open(data, '_blank');
        }
    };
}
